# ToastContainer

Raise a toast notification by triggering an action. You have to add `<ToastContainer/>` in your app at root level and then just use `addToast` function to raise the toast.

```tsx
import { ToastContainer, addToast } from '@medly-components/core';

export const App: React.SFC = () => {
    const handleClick = () => addToast({ variant: 'success', header: 'User Notification', message: 'User successfully added' });

    return (
        <>
            <button onClick={handleClick}>Notify !</button>
            <ToastContainer position="top" />
        </>
    );
};
```
