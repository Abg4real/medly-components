### Raise a toast notification by triggering an action

<Preview>
    <Story name="ToastContainer" decorators={[withKnobs]}>
        <>
            <ToastContainer position={select('Position', positions, 'top')}></ToastContainer>
            <ToastButtons>

                    onClick={() => {
                        addToast({
                            variant: 'info',
                            header: 'Default Notification',
                            message: 'If there is a long description that breaks to multiple lines, then the height of the'
                        });
                    }}
                >
                    Raise toast
                </Button>
                <Button
                    onClick={() => {
                        addToast({
                            icon: PersonAddIcon,

                            header: 'Default Notification',
                            message: 'If there is a long description that breaks to multiple lines, then the height of the'
                        });
                    }}
                >
                    Raise toast with Icon
                </Button>
            </ToastButtons>
        </>
    </Story>
