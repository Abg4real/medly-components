# SearchBox

### Add a auto suggest text box

<Preview withToolbar>
    <Story name="SearchBox">
        {() => {
            const [options, setOptions] = useState([]);
            const onGetOptions = value => {
                if (value) {
                    setOptions(state => {
                        return [...optionsArray];
                    });
                } else {
                    setOptions(state => {
                        return new Array();
                    });
                }
            };
            const onClear = () => {
                setOptions([]);
            };
            return (
                <SearchBox
                    placeholder="Search"
                    onClear={onClear}
                    onChange={onGetOptions}
                    options={options}
                    boxSize={select('boxSize', boxSize, 'S')}
                />
            );
        }}
    </Story>
</Preview>

<Preview withToolbar>
    <Story name="SearchBoxWithoutOnClear">
        {() => {
            const [value, setValue] = useState([]);
            const onGetOptions = value => setValue(value ? [...optionsArray] : []);
            return <SearchBox placeholder="Search" onChange={onGetOptions} options={value} boxSize={select('boxSize', boxSize, 'S')} />;
        }}
    </Story>
</Preview>
