# SearchBox

### Add a auto suggest text box

<Preview withToolbar>
    <Story name="SearchBox">
        {() => {
            const [options, setOptions] = useState([]);
            const onGetOptions = () => {
                setOptions(optionsArray);
            };
            return <SearchBox placeholder="Search" getOptions={onGetOptions} options={options} boxSize={select('boxSize', boxSize, 'S')} />;
        }}
    </Story>
</Preview>
